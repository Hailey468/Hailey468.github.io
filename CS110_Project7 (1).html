plt.ylabel('Average Quality of Sleep')
plt.xticks(rotation=45, ha='right') # Rotate x-axis labels for better readability

plt.show()

# prompt: Take the scatter plot from above and enhance it by providing data labels for the points as well as a legend under the graph that tells which color corresponded to how many responses

# Recreate the scatter plot with data labels and a legend based on the count of responses per occupation
plt.figure(figsize=(12, 7)) # Increased figure size

# To create a legend based on counts, we need to map occupations to colors based on their counts
# Let's calculate the count for each occupation first
occupation_counts = df['Occupation'].value_counts().reset_index()
occupation_counts.columns = ['Occupation', 'Count']

# Merge the counts back to the original dataframe to use it for coloring
df_with_counts = pd.merge(df, occupation_counts, on='Occupation')

# Create the scatter plot, coloring points by count and using the 'Count' column for the legend
scatter = sns.scatterplot(data=df_with_counts, x='Occupation', y='Quality of Sleep', hue='Count', alpha=0.6, s=50) # Increased marker size

plt.title('Quality of Sleep vs. Occupation (Colored by Occupation Count)', fontsize=16) # Title update
plt.xlabel('Occupation', fontsize=12)
plt.ylabel('Quality of Sleep', fontsize=12)
plt.xticks(rotation=45, ha='right', fontsize=10)
plt.grid(axis='y', linestyle='--', alpha=0.7) # Add grid lines

# Add a legend outside the plot
plt.legend(title='Number of Responses', bbox_to_anchor=(1.05, 1), loc='upper left')

plt.tight_layout(rect=[0, 0, 0.85, 1]) # Adjust layout to make space for the legend
plt.show()
